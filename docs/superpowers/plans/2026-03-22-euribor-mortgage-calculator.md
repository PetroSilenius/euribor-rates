# Euribor Mortgage Calculator Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single-page static Astro + React website that helps Finnish homeowners compare Euribor mortgage reference rates (3m / 6m / 12m) and decide whether to switch.

**Architecture:** One React island (`MortgageCalculator.tsx`) owns all state and delegates all math to pure functions in `lib/simulation.ts`. Computed results flow down as props to six presentational child components. The Astro shell is static HTML with no routing.

**Tech Stack:** Astro (static), React (islands via `@astrojs/react`), shadcn/ui (manually installed), Tailwind CSS, TypeScript, recharts

**Spec:** `docs/superpowers/specs/2026-03-22-euribor-mortgage-calculator-design.md`

// ─── Types ───────────────────────────────────────────────────────────────────

export interface SimulationInputs {
  loanAmount: number      // €
  termMonths: number      // e.g. 240 for 20 years
  marginDecimal: number   // e.g. 0.0055 for 0.55%
  currentTenor: 3 | 6 | 12
}

export interface PaymentRow {
  tenor: 3 | 6 | 12
  rate: number            // euriborRate + marginDecimal
  monthlyPayment: number  // €
  annualCost: number      // monthlyPayment * 12
  diffMonthly: number     // this tenor - currentTenor payment (negative = cheaper)
  diffAnnual: number      // diffMonthly * 12
}

export type Scenario = 'flat' | 'rise' | 'fall'

export interface MonthlyPoint {
  month: number           // 0–23
  payment3m: number
  payment6m: number
  payment12m: number
}

export interface BreakevenResult {
  fromTenor: 3 | 6 | 12
  toTenor: 3 | 6 | 12
  monthlySavings: number  // fromPayment - toPayment; positive = toTenor cheaper
  monthsToBreakeven: number | null  // null when monthlySavings <= 0
  switchingFee: number
}

export interface Recommendation {
  verdict: 'stay' | 'marginal' | 'switch'
  reason: string
}

// ─── Constants ───────────────────────────────────────────────────────────────

export const EURIBOR_RATES: Record<3 | 6 | 12, number> = {
  3: 0.0215,
  6: 0.0233,
  12: 0.0252,
}

const TENOR_ORDER: (3 | 6 | 12)[] = [3, 6, 12]

// ─── Helpers ─────────────────────────────────────────────────────────────────

/** Standard annuity formula. Returns 0 if loanAmount is 0. */
export function annuity(P: number, annualRate: number, n: number): number {
  if (P === 0 || n === 0) return 0
  const r = annualRate / 12
  if (r === 0) return P / n
  return (P * r * Math.pow(1 + r, n)) / (Math.pow(1 + r, n) - 1)
}
```

export function computePayments(inputs: SimulationInputs): PaymentRow[] {
  const currentPayment = annuity(
    inputs.loanAmount,
    EURIBOR_RATES[inputs.currentTenor] + inputs.marginDecimal,
    inputs.termMonths,
  )

  return TENOR_ORDER.map(tenor => {
    const rate = EURIBOR_RATES[tenor] + inputs.marginDecimal
    const monthlyPayment = annuity(inputs.loanAmount, rate, inputs.termMonths)
    const annualCost = monthlyPayment * 12
    const diffMonthly = tenor === inputs.currentTenor ? 0 : monthlyPayment - currentPayment
    return { tenor, rate, monthlyPayment, annualCost, diffMonthly, diffAnnual: diffMonthly * 12 }
  })
}
```

```typescript
const RESET_SCHEDULES: Record<3 | 6 | 12, number[]> = {
  3: [0, 3, 6, 9, 12, 15, 18, 21],
  6: [0, 6, 12, 18],
  12: [0, 12],
}

function euriborDelta(scenario: Scenario, month: number): number {
  const t = Math.min(month, 11) / 11
  if (scenario === 'flat') return 0
  if (scenario === 'rise') return t * 0.01
  return t * -0.005
}

function tenorRate(tenor: 3 | 6 | 12, month: number, scenario: Scenario, marginDecimal: number): number {
  return EURIBOR_RATES[tenor] + euriborDelta(scenario, month) + marginDecimal
}

function computeTenorPoints(
  tenor: 3 | 6 | 12,
  inputs: SimulationInputs,
  scenario: Scenario,
): number[] {
  const resets = new Set(RESET_SCHEDULES[tenor])
  let balance = inputs.loanAmount
  let remainingMonths = inputs.termMonths
  let payment = 0
  const payments: number[] = []

  for (let month = 0; month < 24; month++) {
    if (remainingMonths <= 0) {
      payments.push(0)
      continue
    }
    if (resets.has(month)) {
      payment = annuity(balance, tenorRate(tenor, month, scenario, inputs.marginDecimal), remainingMonths)
    }
    payments.push(payment)
    const rate = tenorRate(tenor, month, scenario, inputs.marginDecimal)
    const interest = balance * (rate / 12)
    const principal = payment - interest
    balance -= principal
    remainingMonths -= 1
  }

  return payments
}

export function computeScenario(inputs: SimulationInputs, scenario: Scenario): MonthlyPoint[] {
  const points3m = computeTenorPoints(3, inputs, scenario)
  const points6m = computeTenorPoints(6, inputs, scenario)
  const points12m = computeTenorPoints(12, inputs, scenario)

  return Array.from({ length: 24 }, (_, month) => ({
    month,
    payment3m: points3m[month],
    payment6m: points6m[month],
    payment12m: points12m[month],
  }))
}
```

```typescript
const BREAKEVEN_ORDER: Record<3 | 6 | 12, [3 | 6 | 12, 3 | 6 | 12]> = {
  3:  [6, 12],
  6:  [3, 12],
  12: [6, 3],
}

export function computeBreakeven(inputs: SimulationInputs, switchingFee: number): BreakevenResult[] {
  const [altA, altB] = BREAKEVEN_ORDER[inputs.currentTenor]

  function makeResult(toTenor: 3 | 6 | 12): BreakevenResult {
    const fromPayment = annuity(
      inputs.loanAmount,
      EURIBOR_RATES[inputs.currentTenor] + inputs.marginDecimal,
      inputs.termMonths,
    )
    const toPayment = annuity(
      inputs.loanAmount,
      EURIBOR_RATES[toTenor] + inputs.marginDecimal,
      inputs.termMonths,
    )
    const monthlySavings = fromPayment - toPayment
    const monthsToBreakeven =
      monthlySavings <= 0 ? null : Math.ceil(switchingFee / monthlySavings)
    return {
      fromTenor: inputs.currentTenor,
      toTenor,
      monthlySavings,
      monthsToBreakeven,
      switchingFee,
    }
  }

  return [makeResult(altA), makeResult(altB)]
}

export function computeRecommendation(
  payments: PaymentRow[],
  currentTenor: 3 | 6 | 12,
  scenario: Scenario,
): Recommendation {
  const currentRow = payments.find(r => r.tenor === currentTenor)!
  const alternatives = payments.filter(r => r.tenor !== currentTenor)
  const cheapestAlt = alternatives.reduce((a, b) =>
    a.monthlyPayment < b.monthlyPayment ? a : b,
  )
  const rateGap = cheapestAlt.rate - currentRow.rate

  const Y = Math.abs(Math.round(cheapestAlt.diffMonthly))
  const Z = (Math.abs(rateGap) * 100).toFixed(2) + '%'
  const scenarioLabel = scenario === 'flat' ? 'flat' : 'falling'

  // Rule 1: stay
  if (rateGap >= 0 || scenario === 'rise') {
    if (rateGap >= 0) {
      return {
        verdict: 'stay',
        reason: `${currentTenor}m Euribor is already your cheapest option at current rates.`,
      }
    }
    return {
      verdict: 'stay',
      reason: `${cheapestAlt.tenor}m could save €${Y}/month but rates are rising — your ${currentTenor}m lock looks sensible.`,
    }
  }

  // Rule 2: switch
  if (rateGap < -0.002) {
    return {
      verdict: 'switch',
      reason: `${cheapestAlt.tenor}m is ${Z} cheaper than your current tenor. At ${scenarioLabel} rates, switching makes sense.`,
    }
  }

  // Rule 3: marginal
  return {
    verdict: 'marginal',
    reason: `${cheapestAlt.tenor}m is marginally cheaper (${Z}), but the saving is small — depends on your risk tolerance.`,
  }
}
```

// src/components/LoanInputPanel.tsx
import { useState } from 'react'
import { Slider } from '@/components/ui/slider'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'
import { Tabs, TabsList, TabsTrigger } from '@/components/ui/tabs'
import { cn } from '@/lib/utils'

interface LoanInputPanelProps {
  loanAmount: number
  termYears: number
  margin: number
  currentTenor: 3 | 6 | 12
  onLoanAmountChange: (v: number) => void
  onTermYearsChange: (v: number) => void
  onMarginChange: (v: number) => void
  onCurrentTenorChange: (v: 3 | 6 | 12) => void
}

function clamp(value: number, min: number, max: number): number {
  return Math.min(max, Math.max(min, value))
}

interface SliderInputFieldProps {
  label: string
  value: number
  min: number
  max: number
  step: number
  format: (v: number) => string
  onChange: (v: number) => void
}

function SliderInputField({ label, value, min, max, step, format, onChange }: SliderInputFieldProps) {
  const [draft, setDraft] = useState<string | null>(null)

  function handleBlur() {
    if (draft === null) return
    const parsed = parseFloat(draft.replace(',', '.'))
    if (!isNaN(parsed)) {
      onChange(clamp(parsed, min, max))
    }
    setDraft(null)
  }

  return (
    <div className="space-y-2">
      <div className="flex items-center justify-between">
        <Label>{label}</Label>
        <Input
          className="w-28 text-right h-7 text-sm"
          value={draft ?? format(value)}
          onChange={e => setDraft(e.target.value)}
          onBlur={handleBlur}
          onFocus={e => setDraft(String(value))}
        />
      </div>
      <Slider
        min={min}
        max={max}
        step={step}
        value={[value]}
        onValueChange={([v]) => onChange(v)}
      />
    </div>
  )
}

export function LoanInputPanel({
  loanAmount, termYears, margin, currentTenor,
  onLoanAmountChange, onTermYearsChange, onMarginChange, onCurrentTenorChange,
}: LoanInputPanelProps) {
  return (
    <div className="space-y-6">
      <SliderInputField
        label="Lainan määrä"
        value={loanAmount}
        min={10_000}
        max={600_000}
        step={5_000}
        format={v => `€${v.toLocaleString('fi-FI')}`}
        onChange={onLoanAmountChange}
      />
      <SliderInputField
        label="Jäljellä oleva laina-aika (vuotta)"
        value={termYears}
        min={1}
        max={30}
        step={1}
        format={v => `${v} v`}
        onChange={onTermYearsChange}
      />
      <SliderInputField
        label="Marginaali"
        value={margin}
        min={0.1}
        max={3.0}
        step={0.05}
        format={v => `${v.toFixed(2)}%`}
        onChange={onMarginChange}
      />
      <div className="space-y-2">
        <Label>Nykyinen viitekorko</Label>
        <Tabs
          value={String(currentTenor)}
          onValueChange={v => onCurrentTenorChange(Number(v) as 3 | 6 | 12)}
        >
          <TabsList className="w-full">
            {([3, 6, 12] as const).map(t => (

          </TabsList>
